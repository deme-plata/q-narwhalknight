\subsection{2.2 OAuth2 Authentication (For Client-Side / Third-Party Apps)}

\textbf{Use Case:} When building a wallet or dApp that needs user authorization.

\textbf{OAuth2 Flow:}
\begin{enumerate}
    \item Redirect user to: \\\texttt{https://quillon.xyz/auth/oauth/authorize?client\_id=YOUR\_ID}
    \item User approves scopes (e.g., \texttt{privacy:mix}, \texttt{privacy:tor-relay})
    \item Receive authorization code via redirect
    \item Exchange code for access token
    \item Use access token in \texttt{Authorization} header
\end{enumerate}

\textbf{Example (JavaScript):}
\begin{lstlisting}[language=JavaScript]
// Step 1: Redirect to authorization
const CLIENT_ID = "your_client_id";
const REDIRECT_URI = "https://yourapp.com/callback";
const authUrl = `https://quillon.xyz/auth/oauth/authorize?` +
  `client_id=${CLIENT_ID}&` +
  `redirect_uri=${encodeURIComponent(REDIRECT_URI)}&` +
  `scope=privacy:mix privacy:tor-relay&` +
  `response_type=code`;

window.location.href = authUrl;

// Step 2: Exchange authorization code for token (in callback)
async function exchangeCodeForToken(code) {
  const response = await fetch(
    'https://quillon.xyz/auth/oauth/token',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        grant_type: 'authorization_code',
        code: code,
        client_id: CLIENT_ID,
        client_secret: CLIENT_SECRET, // Keep server-side!
        redirect_uri: REDIRECT_URI
      })
    }
  );

  const { access_token } = await response.json();
  return access_token;
}

// Step 3: Use access token
async function mixTransaction(txHex, accessToken) {
  const response = await fetch(
    'https://quillon.xyz/api/v1/privacy/mix/submit',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        chain: 'ethereum',
        transaction_hex: txHex,
        privacy_level: 'standard'
      })
    }
  );

  return response.json();
}
\end{lstlisting}

const { ethers } = require('ethers');
const axios = require('axios');

const API_KEY = "your_api_key";
const BASE_URL = "https://quillon.xyz";

async function privateUniswapSwap(
  tokenIn,      // e.g., WETH address
  tokenOut,     // e.g., USDC address
  amountIn,     // Amount in Wei
  minAmountOut, // Minimum output (slippage protection)
  wallet        // ethers.Wallet instance
) {
  // Step 1: Build Uniswap swap transaction
  const uniswapRouter = "0x7a250d5630B4cF539739dF2C5dAcb4c659F2488D";
  const abi = [
    "function swapExactTokensForTokens(uint amountIn, " +
    "uint amountOutMin, address[] path, address to, uint deadline)"
  ];

  const router = new ethers.Contract(uniswapRouter, abi, wallet);

  const tx = await router.populateTransaction.swapExactTokensForTokens(
    amountIn,
    minAmountOut,
    [tokenIn, tokenOut],
    wallet.address,
    Math.floor(Date.now() / 1000) + 60 * 20 // 20 min deadline
  );

  // Step 2: Sign transaction
  const signedTx = await wallet.signTransaction(tx);

  // Step 3: Submit to Q-NarwhalKnight private relay
  const response = await axios.post(
    `${BASE_URL}/api/v1/privacy/ethereum/mev-protect`,
    {
      signed_transaction: signedTx,
      max_block_number: null,  // No expiry
      options: {
        tor_relay: true,          // Hide IP
        flashbots_relay: true,    // Private mempool
        simulate: true,           // Pre-execution simulation
        require_success: true     // Revert if unprofitable
      }

    {
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json',
        'Idempotency-Key': ethers.utils.id(signedTx)  // Prevent duplicates
      }
    }
  );

  const result = response.data;

  if (result.success) {
    return {
      transactionHash: result.data.transaction_hash,
      mevProtected: true,
      estimatedSavings: result.data.estimated_mev_savings_usd,
      blockNumber: result.data.included_in_block
    };
  } else {
    throw new Error(`MEV protection failed: ${result.error}`);
  }
}

\section{12. Advanced Features}

\subsection{12.1 Idempotency and Error Handling}

\textbf{Why Idempotency Matters:}
\begin{itemize}
    \item Network failures can cause retries
    \item Without idempotency, you might pay twice
    \item We cache responses for 24 hours
\end{itemize}

\textbf{Best Practice:}
\begin{lstlisting}[language=python]
import uuid
import requests

def idempotent_request(endpoint, payload):
    """
    Make an idempotent API request.
    Safe to retry without double-charging.
    """

    # Generate unique idempotency key (deterministic based on payload)
    idempotency_key = str(uuid.uuid5(
        uuid.NAMESPACE_DNS,
        f"myapp:{endpoint}:{json.dumps(payload, sort_keys=True)}"
    ))

    max_retries = 3

    for attempt in range(max_retries):
        try:
            response = requests.post(
                f"{BASE_URL}{endpoint}",
                headers={
                    "Authorization": f"Bearer {API_KEY}",
                    "Content-Type": "application/json",
                    "Idempotency-Key": idempotency_key
                },
                json=payload,
                timeout=30
            )

            if response.status_code == 409:
                # Conflict: Idempotency key reused with different body
                raise ValueError("Request body changed, use new key")

            response.raise_for_status()
            return response.json()

        except requests.exceptions.Timeout:
            if attempt == max_retries - 1:
                raise
            time.sleep(2 ** attempt)  # Exponential backoff

        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 429:
                # Rate limited
                retry_after = int(e.response.headers.get('Retry-After', 60))
                time.sleep(retry_after)
            else:
                raise

# Example usage
result = idempotent_request("/api/v1/privacy/mix/submit", {
    "chain": "bitcoin",
    "transaction_hex": "...",
    "privacy_level": "maximum"
})
\end{lstlisting}

\subsection{13.1 Security Checklist}

\begin{itemize}
    \item[$\square$] \textbf{Never expose API keys in client-side code} - Use backend proxy or OAuth2

    \item[$\square$] \textbf{Use HTTPS only} - Never send requests over HTTP
    \item[$\square$] \textbf{Log API errors securely} - Don't log full request/response (may contain keys)
\end{itemize}

import asyncio
import aiohttp

async def parallel_mixing(transactions):
    """
    Mix multiple transactions in parallel for maximum throughput.
    """

    async with aiohttp.ClientSession() as session:
        tasks = []

        for tx in transactions:
            task = mix_transaction_async(session, tx)
            tasks.append(task)

        # Execute all mixing requests concurrently
        results = await asyncio.gather(*tasks, return_exceptions=True)

\subsection{13.3 Monitoring and Alerting}

\begin{lstlisting}[language=python]
import time
import logging

class PaaSClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.logger = logging.getLogger('paas_client')

        # Track metrics
        self.requests_sent = 0
        self.requests_failed = 0
        self.avg_latency_ms = 0

    def mix_transaction(self, tx):
        start = time.time()

        try:
            response = self._make_request("/api/v1/privacy/mix/submit", tx)

            latency = (time.time() - start) * 1000
            self._update_metrics(success=True, latency=latency)

            if latency > 1000:  # Slow request
                self.logger.warning(f"Slow mixing request: {latency}ms")

            return response

        except Exception as e:
            self._update_metrics(success=False)
            self.logger.error(f"Mixing failed: {str(e)}")

            # Alert on-call engineer if error rate > 5%
            if self.requests_failed / self.requests_sent > 0.05:
                self._send_alert("High PaaS error rate!")

            raise

    def _update_metrics(self, success, latency=0):
        self.requests_sent += 1
        if not success:
            self.requests_failed += 1
        if latency > 0:
            self.avg_latency_ms = (
                (self.avg_latency_ms * (self.requests_sent - 1) + latency)
                / self.requests_sent
            )
\end{lstlisting}

\toprule
\textbf{Code} & \textbf{Solution} \\
\midrule

400 & \textbf{Bad Request:} Check request body format. Ensure all required fields present. \\
401 & \textbf{Unauthorized:} Invalid API key. Verify key is correct and not expired. \\
402 & \textbf{Payment Required:} Insufficient balance. Top up your QUG balance. \\
409 & \textbf{Conflict:} Idempotency key reused with different body. Use new key or same body. \\
429 & \textbf{Rate Limited:} Too many requests. Check \texttt{Retry-After} header and slow down. \\
500 & \textbf{Server Error:} Our fault. Retry with exponential backoff. Contact support if persists. \\
503 & \textbf{Service Unavailable:} Temporary downtime. Check status page: \texttt{status.q-narwhalknight.io} \\
