\begin{abstract}
This paper analyzes the storage growth problem in Q-NarwhalKnight blockchain, where 7 days of operation generates 4GB of data (110,000 blocks). We review multiple pruning strategies including state pruning, block pruning, and hybrid approaches. Our analysis shows that aggressive pruning can reduce storage requirements by 85-95\% while maintaining security guarantees. We propose a tiered storage architecture with configurable retention policies optimized for different node types (full nodes, archive nodes, light clients).
\end{abstract}

\section{Introduction}

\subsection{Problem Statement}

Q-NarwhalKnight currently exhibits the following storage characteristics:

\begin{equation}
\text{Storage}_{\text{7 days}} = 4\text{ GB} = \frac{4 \times 10^9 \text{ bytes}}{110,000 \text{ blocks}} \approx 36.4\text{ KB/block}
\end{equation}

Extrapolating to one year:

\begin{equation}
\text{Storage}_{\text{1 year}} = \frac{4\text{ GB}}{7\text{ days}} \times 365\text{ days} \approx 208\text{ GB/year}
\end{equation}

\textbf{Current Growth Rate:} 208 GB/year is unsustainable for consumer hardware nodes.

\section{Pruning Strategies}

\subsection{Strategy 1: State Pruning}

\textbf{Concept:} Store only the current state, discard historical state snapshots.

\subsubsection{Implementation}

\begin{algorithm}[H]
\caption{State Pruning Algorithm}
\begin{algorithmic}
\REQUIRE Block height $h$, retention depth $d$
\STATE $\text{current\_state} \leftarrow \text{compute\_state}(h)$
\FOR{each block $b$ in range $[h - d, h)$}
    \STATE Keep $\text{state}(b)$ \COMMENT{Recent state kept}
\ENDFOR
\FOR{each block $b < h - d$}
    \STATE Delete $\text{state}(b)$ \COMMENT{Old state pruned}
    \STATE Keep $\text{header}(b)$ \COMMENT{Headers always kept}
\ENDFOR
\end{algorithmic}
\end{algorithm}

\textbf{Storage Savings:}
\begin{equation}
\text{Savings}_{\text{state}} = 55\% \times \left(1 - \frac{d}{h}\right)
\end{equation}

For $d = 1000$ blocks (retention of last ~3 hours at current block rate):
\begin{equation}
\text{Savings}_{\text{state}} = 55\% \times \left(1 - \frac{1000}{110000}\right) \approx 54.5\%
\end{equation}

\subsubsection{Security Trade-offs}

\begin{itemize}
\item \textbf{Pros:}
    \begin{itemize}
    \item Massive storage reduction (54.5\%)
    \item State always reconstructable from recent blocks
    \item No impact on chain verification
    \end{itemize}
\item \textbf{Cons:}
    \begin{itemize}
    \item Historical balance queries impossible
    \item Re-org handling requires state reconstruction
    \item Light clients cannot independently verify old state
    \end{itemize}
\end{itemize}

\subsection{Strategy 2: Block Pruning}

\textbf{Concept:} Store only block headers + recent full blocks, discard old block bodies.

\subsubsection{Implementation}

\begin{lstlisting}[language=Rust]
// Configurable pruning policy
pub struct PruningConfig {
    /// Keep full blocks for last N blocks
    pub full_block_retention: u64,
    /// Keep headers forever (small)
    pub header_retention: u64, // u64::MAX
    /// Keep state for last M blocks
    pub state_retention: u64,
}

impl Default for PruningConfig {
    fn default() -> Self {
        Self {
            full_block_retention: 10_000, // ~1 day
            header_retention: u64::MAX,
            state_retention: 1_000,
        }
    }
}
\end{lstlisting}

\textbf{Storage Savings:}
\begin{equation}
\text{Savings}_{\text{blocks}} = (100\% - 1.4\%) \times \left(1 - \frac{10000}{110000}\right) \approx 89.6\%
\end{equation}

\subsection{Strategy 3: Hybrid Tiered Pruning (Recommended)}

\textbf{Concept:} Different retention policies for different data types.

\begin{table}[h]
\centering
\begin{tabular}{lrr}
\toprule
\textbf{Data Type} & \textbf{Retention} & \textbf{Rationale} \\
\midrule
Block Headers & Forever & Tiny (512 bytes) \\
Recent Full Blocks & 10,000 blocks & Re-org safety \\
Recent State & 1,000 blocks & Quick queries \\
Old Block Bodies & Pruned & Reconstruct from headers \\
Old State & Pruned & Reconstruct if needed \\
Quantum Metadata & 10,000 blocks & Scientific value \\
\bottomrule
\end{tabular}
\caption{Tiered retention policy}
\end{table}

\subsubsection{Total Storage Calculation}

\begin{align}
\text{Storage}_{\text{pruned}} &= \underbrace{h \times 512\text{ B}}_{\text{All headers}} + \underbrace{d_{\text{full}} \times 36.4\text{ KB}}_{\text{Recent full blocks}} \nonumber \\
&\quad + \underbrace{d_{\text{state}} \times 20\text{ KB}}_{\text{Recent state}} \\
&= 110000 \times 512 + 10000 \times 36400 + 1000 \times 20000 \nonumber \\
&= 56.3\text{ MB} + 364\text{ MB} + 20\text{ MB} \nonumber \\
&= \mathbf{440.3\text{ MB}}
\end{align}

\textbf{Compared to unpruned:}
\begin{equation}
\text{Reduction} = \frac{4000\text{ MB} - 440\text{ MB}}{4000\text{ MB}} \times 100\% = \mathbf{89\%}
\end{equation}

\section{Implementation Architecture}

\subsection{RocksDB Column Family Structure}

\begin{lstlisting}[language=Rust]
pub struct PrunedStorage {
    // Separate column families for different retention
    cf_headers: ColumnFamily,      // Never pruned
    cf_recent_blocks: ColumnFamily, // 10k retention
    cf_recent_state: ColumnFamily,  // 1k retention
    cf_indices: ColumnFamily,       // Compact indices

    pruning_config: PruningConfig,
    last_prune_height: AtomicU64,
}

impl PrunedStorage {
    pub async fn prune_old_data(&self, current_height: u64)
        -> Result<PruningStats>
    {
        let mut stats = PruningStats::default();

        // Prune old full blocks
        let prune_blocks_below = current_height
            .saturating_sub(self.pruning_config.full_block_retention);

        for height in self.last_prune_height..prune_blocks_below {
            // Keep header, delete body
            self.delete_block_body(height).await?;
            stats.blocks_pruned += 1;
        }

        // Prune old state
        let prune_state_below = current_height
            .saturating_sub(self.pruning_config.state_retention);

        for height in self.last_prune_height..prune_state_below {
            self.delete_state_snapshot(height).await?;
            stats.state_pruned += 1;
        }

        self.last_prune_height.store(
            current_height,
            Ordering::Relaxed
        );

        // RocksDB compaction to reclaim space
        self.compact_range(None, None).await?;

        Ok(stats)
    }
}
\end{lstlisting}

\subsection{Automatic Pruning Scheduler}

\begin{algorithm}[H]
\caption{Continuous Pruning Daemon}
\begin{algorithmic}
\REQUIRE Pruning interval $I$ (e.g., 1000 blocks)
\WHILE{node running}
    \STATE $h \leftarrow \text{current\_height}()$
    \IF{$h \bmod I = 0$}
        \STATE $\text{stats} \leftarrow \text{prune\_old\_data}(h)$
        \STATE Log pruning statistics
        \STATE Update Prometheus metrics
    \ENDIF
    \STATE Sleep for 10 seconds
\ENDWHILE
\end{algorithmic}
\end{algorithm}

\section{Performance Analysis}

\subsection{Pruning Overhead}

\textbf{Deletion Rate:}
\begin{equation}
\text{Deletions/day} = \frac{110000\text{ blocks}}{7\text{ days}} - 10000 = 5714\text{ blocks/day}
\end{equation}

\textbf{I/O Cost per Pruning Cycle:}
\begin{itemize}
\item Delete 1000 block bodies: $\sim$1000 RocksDB delete ops
\item Delete 1000 state snapshots: $\sim$1000 RocksDB delete ops
\item Compaction: $\sim$100ms (amortized)
\end{itemize}

\textbf{Total Overhead:} $<$ 1\% of CPU time, $<$ 5\% of I/O bandwidth.

\subsection{Re-org Handling}

With 10,000 block retention:

\begin{equation}
\text{Re-org Safety} = \frac{10000\text{ blocks}}{15716\text{ blocks/day}} \approx 0.64\text{ days} = 15.3\text{ hours}
\end{equation}

\textbf{Analysis:} Bitcoin's longest re-org was 24 blocks (4 hours). Our 15-hour safety margin is adequate.

\section{Sync Performance Impact}

\subsection{Initial Sync with Pruning}

\textbf{Naive Approach (Slow):}
\begin{algorithmic}
\FOR{each block $b$ from genesis}
    \STATE Download block $b$
    \STATE Process block $b$
    \STATE \textbf{Prune if $b < h - d$}  \COMMENT{Extra I/O!}
\ENDFOR
\end{algorithmic}

\textbf{Optimized Approach (Fast):}
\begin{algorithmic}
\STATE $h \leftarrow \text{network\_height}()$
\STATE $\text{skip\_to} \leftarrow \max(1, h - d)$
\FOR{each block $b$ from $\text{skip\_to}$ to $h$}
    \STATE Download block $b$
    \STATE Process block $b$
\ENDFOR
\STATE Sync only recent $d$ blocks!
\end{algorithmic}

\section{Security Considerations}

\subsection{Data Availability}

\textbf{Concern:} If all nodes prune, historical data may be lost.

\textbf{Solution:} Incentivize archive nodes via:
\begin{itemize}
\item Mining rewards for serving historical blocks
\item Validator requirements (must run archive mode)
\item Community-run archives (explorers, indexers)
\end{itemize}

\subsection{State Reconstruction}

Even with pruning, any state can be reconstructed:

\begin{algorithm}[H]
\caption{State Reconstruction}
\begin{algorithmic}
\REQUIRE Target state at height $h_{\text{target}}$, current height $h_{\text{current}}$
\IF{$h_{\text{target}} \geq h_{\text{current}} - d$}
    \RETURN cached state \COMMENT{Within retention}
\ELSE
    \STATE $\text{state} \leftarrow \text{genesis\_state}()$
    \FOR{block $b$ from genesis to $h_{\text{target}}$}
        \STATE Download header of $b$ from archive node
        \STATE Apply state transitions
    \ENDFOR
    \RETURN state
\ENDIF
\end{algorithmic}
\end{algorithm}

\subsection{Fraud Proofs}

Light nodes can verify claims via fraud proofs:

\begin{equation}
\text{Proof}_{\text{fraud}} = \{\text{block\_header}, \text{Merkle\_path}, \text{state\_pre}, \text{state\_post}\}
\end{equation}

Size: $\sim$5 KB (compact even for pruned nodes)

\section{Conclusion}

\begin{itemize}
